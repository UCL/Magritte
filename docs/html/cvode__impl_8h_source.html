280 /*
281  * -----------------------------------------------------------------
282  * Communication between CVODE and a CVODE Linear Solver
283  * -----------------------------------------------------------------
284  * convfail (input to cv_lsetup)
285  *
286  * CV_NO_FAILURES : Either this is the first cv_setup call for this
287  * step, or the local error test failed on the
288  * previous attempt at this step (but the Newton
289  * iteration converged).
290  *
291  * CV_FAIL_BAD_J : This value is passed to cv_lsetup if
292  *
293  * (a) The previous Newton corrector iteration
294  * did not converge and the linear solver's
295  * setup routine indicated that its Jacobian-
296  * related data is not current
297  * or
298  * (b) During the previous Newton corrector
299  * iteration, the linear solver's solve routine
300  * failed in a recoverable manner and the
301  * linear solver's setup routine indicated that
302  * its Jacobian-related data is not current.
303  *
304  * CV_FAIL_OTHER : During the current internal step try, the
305  * previous Newton iteration failed to converge
306  * even though the linear solver was using current
307  * Jacobian-related data.
308  * -----------------------------------------------------------------
309  */
310 
311 /* Constants for convfail (input to cv_lsetup) */
312 
313 #define CV_NO_FAILURES 0
314 #define CV_FAIL_BAD_J 1
315 #define CV_FAIL_OTHER 2

317 /*
318  * -----------------------------------------------------------------
319  * int (*cv_linit)(CVodeMem cv_mem);
320  * -----------------------------------------------------------------
321  * The purpose of cv_linit is to complete initializations for a
322  * specific linear solver, such as counters and statistics.
323  * An LInitFn should return 0 if it has successfully initialized the
324  * CVODE linear solver and a negative value otherwise.
325  * If an error does occur, an appropriate message should be sent to
326  * the error handler function.
327  * -----------------------------------------------------------------
328  */
329 
330 /*
331  * -----------------------------------------------------------------
332  * int (*cv_lsetup)(CVodeMem cv_mem, int convfail, N_Vector ypred,
333  * N_Vector fpred, booleantype *jcurPtr,
334  * N_Vector vtemp1, N_Vector vtemp2,
335  * N_Vector vtemp3);
336  * -----------------------------------------------------------------
337  * The job of cv_lsetup is to prepare the linear solver for
338  * subsequent calls to cv_lsolve. It may recompute Jacobian-
339  * related data is it deems necessary. Its parameters are as
340  * follows:
341  *
342  * cv_mem - problem memory pointer of type CVodeMem. See the
343  * typedef earlier in this file.
344  *
345  * convfail - a flag to indicate any problem that occurred during
346  * the solution of the nonlinear equation on the
347  * current time step for which the linear solver is
348  * being used. This flag can be used to help decide
349  * whether the Jacobian data kept by a CVODE linear
350  * solver needs to be updated or not.
351  * Its possible values have been documented above.
352  *
353  * ypred - the predicted y vector for the current CVODE internal
354  * step.
355  *
356  * fpred - f(tn, ypred).
357  *
358  * jcurPtr - a pointer to a boolean to be filled in by cv_lsetup.
359  * The function should set *jcurPtr=SUNTRUE if its Jacobian
360  * data is current after the call and should set
361  * *jcurPtr=SUNFALSE if its Jacobian data is not current.
362  * Note: If cv_lsetup calls for re-evaluation of
363  * Jacobian data (based on convfail and CVODE state
364  * data), it should return *jcurPtr=SUNTRUE always;
365  * otherwise an infinite loop can result.
366  *
367  * vtemp1 - temporary N_Vector provided for use by cv_lsetup.
368  *
369  * vtemp3 - temporary N_Vector provided for use by cv_lsetup.
370  *
371  * vtemp3 - temporary N_Vector provided for use by cv_lsetup.
372  *
373  * The cv_lsetup routine should return 0 if successful, a positive
374  * value for a recoverable error, and a negative value for an
375  * unrecoverable error.
376  * -----------------------------------------------------------------
377  */
378 
379 /*
380  * -----------------------------------------------------------------
381  * int (*cv_lsolve)(CVodeMem cv_mem, N_Vector b, N_Vector weight,
382  * N_Vector ycur, N_Vector fcur);
383  * -----------------------------------------------------------------
384  * cv_lsolve must solve the linear equation P x = b, where
385  * P is some approximation to (I - gamma J), J = (df/dy)(tn,ycur)
386  * and the RHS vector b is input. The N-vector ycur contains
387  * the solver's current approximation to y(tn) and the vector
388  * fcur contains the N_Vector f(tn,ycur). The solution is to be
389  * returned in the vector b. cv_lsolve returns a positive value
390  * for a recoverable error and a negative value for an
391  * unrecoverable error. Success is indicated by a 0 return value.
392  * -----------------------------------------------------------------
393  */
394 
395 /*
396  * -----------------------------------------------------------------
397  * int (*cv_lfree)(CVodeMem cv_mem);
398  * -----------------------------------------------------------------
399  * cv_lfree should free up any memory allocated by the linear
400  * solver. This routine is called once a problem has been
401  * completed and the linear solver is no longer needed. It should
402  * return 0 upon success, nonzero on failure.
403  * -----------------------------------------------------------------
404  */
